Demonstrate scalability with this Node.js example using clustering for multi-core utilization:

<CodeGroup tabs="Node.js,Docker">
  ```javascript
  const cluster = require('cluster');
  const numCPUs = require('os').cpus().length;

  if (cluster.isMaster) {
    console.log(`Master ${process.pid} is running`);
    for (let i = 0; i < numCPUs; i++) {
      cluster.fork();
    }
  } else {
    // Workers serve requests
    require('http').createServer((req, res) => {
      res.writeHead(200);
      res.end('Scalable Vidiemme app!');
    }).listen(8000);
  }
  ```
  ```dockerfile
  FROM node:18-alpine
  WORKDIR /app
  COPY package*.json ./
  RUN npm install
  COPY . .
  EXPOSE 8000
  CMD ["node", "server.js"]
  ```
</CodeGroup>
